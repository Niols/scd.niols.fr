\begin{dance}
  {The Mall in St. James’s Park}
  {S32 3/3L}{32-bar strathspey for three couples in a three-couple longwise set}

  \phrase{1}{8}{
    1st couple set, cross right hand, cast off one place, advance in the middle
    and pull right shoulder to face first corners. 2nd couple step up on bars
    3-4.
  }

  \phrase{9}{16}{
    1st couple turns first corners both hands, pass each other right shoulder,
    turn second corners both hands and pass each other right shoulder to finish
    on own sides.
  }

  \phrase{17}{24}{
    2nd, 1st and 3rd couples dance parallel reels of three in six bars on the
    lines, 1st man giving left shoulder to 3rd man and 1st woman giving right
    shoulder to 3rd woman. On bars 23-24, 1st and 3rd men exchange place left
    shoulder while 1st and 3rd women exchange place right shoulder.
  }

  \phrase{25}{32}{
    3rd and 1st couples dance the Poussette.
  }

  \repeatfromnewpos

  Devised by \deviser{Nicolas “Niols” Jeannerod} in 2018. The name was suggested
  by Christophe Schuwey in reference to the painting of the same name by Thomas
  Gainsborough.

  Music by \composer{Nicolas “Niols” Jeannerod}.
\end{dance}
